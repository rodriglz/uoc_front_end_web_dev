    if (n1<n2) {

            llistNums.push(index);
        }
                /* aqui omplim array de numeros parells entre 1 i 2 */
        for (var index = 0; index < llistNums.length; index++) {
            var numero = llistNums[index];
            if (numero % 2 === 0) {
                llistParells.push(numero); 
            }
        return llistParells;
        }
    } else {
        for(var index=n2;index<=n1;index++) {
            llistNums.push(index);

        for (var index = 0; index < llistNums.length; index++) {
            var numero = llistNums[index];
            if (numero % 2 === 0) {

            }
        }

    }
}
